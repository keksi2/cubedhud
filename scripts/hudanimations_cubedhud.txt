//Health Bonus Pulse
event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage 	Alpha		"255"		Linear 0.0 0.2
	Animate	PlayerStatusHealthBonusImage 	Alpha		"0"			Linear 0.2 0.4
	
	Animate	PlayerStatusHealthValue		FgColor	"Health Overhealed 1"	Linear	0.0	0.1
	Animate	PlayerStatusHealthValue		FgColor	"Health Overhealed 2"	Linear	0.3 0.4
	Animate	PlayerStatusHealthValueBG	BgColor	"Health Overhealed 1"	Linear	0.0	0.1
	Animate	PlayerStatusHealthValueBG	BgColor	"Health Overhealed 2"	Linear	0.3 0.4

	RunEvent HudHealthBonusPulseLoop	0.25
}

// call to loop HudHealthBonusPulse
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	Animate	PlayerStatusHealthValue		FgColor	"Health Number"	Linear	0.0 0.0
	Animate	PlayerStatusHealthValueBG	BgColor	"Blank"			Linear	0.0 0.0
	
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
}
